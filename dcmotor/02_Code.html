#include <AFMotor.h>
AF_DCMotor motor1(1);
AF_DCMotor motor2(2);
AF_DCMotor motor3(3);
AF_DCMotor motor4(4);

int speed = 150;

void setup() {
  Serial.begin(9600);
  Serial.println("\nModule1.1: Control Car Movement");
  delay(2500);
}

void loop() {
  moveForward();
  moveBackward();
  spinLeft();
  spinRight();
}

void moveForward() {
  // Move the car forward
  motor1.run(FORWARD);
  motor1.setSpeed(speed);
  motor2.run(FORWARD);
  motor2.setSpeed(speed);
  motor3.run(FORWARD);
  motor3.setSpeed(speed);
  motor4.run(FORWARD);
  motor4.setSpeed(speed);
  delay(500);
  // Stop all the wheels from spinning
  stopAllWheels();
}

void moveBackward() {
  // Move the car backward
  motor1.run(BACKWARD);
  motor1.setSpeed(speed);
  motor2.run(BACKWARD);
  motor2.setSpeed(speed);
  motor3.run(BACKWARD);
  motor3.setSpeed(speed);
  motor4.run(BACKWARD);
  motor4.setSpeed(speed);
  delay(500);
  // Stop all the wheels from spinning
  stopAllWheels();
}

void spinLeft() {
  // Move the car to the left
  motor1.run(FORWARD);
  motor1.setSpeed(speed);
  motor2.run(FORWARD);
  motor2.setSpeed(speed);
  motor3.run(BACKWARD);
  motor3.setSpeed(speed);
  motor4.run(BACKWARD);
  motor4.setSpeed(speed);
  delay(500);
  // Stop all the wheels from spinning
  stopAllWheels();
}

void spinRight() {
  // Move the car to the right
  motor1.run(BACKWARD);
  motor1.setSpeed(speed);
  motor2.run(BACKWARD);
  motor2.setSpeed(speed);
  motor3.run(FORWARD);
  motor3.setSpeed(speed);
  motor4.run(FORWARD);
  motor4.setSpeed(speed);
  delay(500);
  // Stop all the wheels from spinning
  stopAllWheels();
}

void stopAllWheels() {
  motor1.run(RELEASE);
  motor2.run(RELEASE);
  motor3.run(RELEASE);
  motor4.run(RELEASE);
  delay(2500);
}
